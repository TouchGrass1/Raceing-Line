FUNCTION FindBoundary(startNode)
    boundarySet = [startNode]
    currentNode = startNode
    foundNext = True

    WHILE foundNext == True
        foundNext = False
        
        // Iterate through expanding search windows (3x3 to 11x11)
        FOR radius = 1 TO 5
            windowSize = (2 * radius) + 1
            neighbors = GetPixelsInWindow(currentNode, windowSize)

            FOR EACH pixel IN neighbors
                IF pixel.colour == GRAY AND NOT(pixel IN boundarySet) THEN
                    IF radius > 1 THEN
                        DrawLine(currentNode, pixel) // Bridge gaps
                    ENDIF
                    
                    boundarySet.append(pixel)
                    currentNode = pixel
                    foundNext = True
                    EXIT FOR // Break neighbor loop
                ENDIF
            NEXT pixel

            IF foundNext == True THEN 
                EXIT FOR // Break radius loop
            ENDIF
        NEXT radius

        // Check if circuit is complete
        IF currentNode == startNode THEN
            foundNext = False
        ENDIF
    ENDWHILE

    RETURN boundarySet
ENDFUNCTION

Genetic Algorithm

// Function to set up intial population
FUNCTION InitialisePopulation(populationSize)
    population = []
    count = 0
    
    WHILE count < populationSize
        path = CreateRandomBSpline()        
        path.time = FindTrackTime(path)
        population.append(path)
        count = count + 1
    ENDWHILE
    
    RETURN population
ENDFUNCTION

// Sort Population by fastest first
FUNCTION EvaluatePopulation(population)
    SORT population BY ascending path.time
    RETURN population
ENDFUNCTION

// Pick random parents, with bias towards faster tracks
FUNCTION SelectParents(population)
    totalFitness = 0
    fitnessValues = []
    
    FOR EACH individual IN population
        fitness = 1 / individual.time
        fitnessValues.append(fitness)
        totalFitness = totalFitness + fitness
    NEXT individual
    
    // Choose parents based on probability (Fitness / TotalFitness)
    parent1 = RandomChoice(population, fitnessValues, totalFitness)
    parent2 = RandomChoice(population, fitnessValues, totalFitness)
    
    RETURN [parent1, parent2]
ENDFUNCTION

// Combines two paths at a random splice point to create a new offspring
FUNCTION Crossover(parent1, parent2)
    splitPoint = RandomFloat(0.25, 0.75)
    index = Floor(length(parent1) * splitPoint)
    
    // Create child by taking the first section of P1 and the second of P2
    childFirstPart = parent1[0 TO index]
    childSecondPart = parent2[index + 1 TO length(parent2) - 1]
    
    child = Combine(childFirstPart, childSecondPart)
    RETURN child
ENDFUNCTION

// Introduces random variation to prevent the population from falling into a local minima
FUNCTION Mutate(individual, mutationRate)
    FOR i = 0 TO length(individual) - 1
        IF Random(0, 1) < mutationRate THEN
            // Replace a node/gene with a random value from the valid track mesh, in the same row
            individual[i] = RandomChoiceFromGenePool()
        ENDIF
    NEXT i
    
    RETURN individual
ENDFUNCTION

FUNCTION Main()
    populationSize = 100
    generations = 100
    eliteRate = 0.1
    mutationRate = 0.1
    
    currentGeneration = InitialisePopulation(populationSize)
    genCount = 0
    
    WHILE genCount < generations
        currentGeneration = EvaluatePopulation(currentGeneration)
        
        // Save top 10% to next generation
        numElites = populationSize * eliteRate
        newPopulation = currentGeneration[0 TO numElites]
        
        // Fill the rest of the population with new offspring
        WHILE length(newPopulation) < populationSize
            parents = SelectParents(currentGeneration)
            child = Crossover(parents[0], parents[1])
            child = Mutate(child, mutationRate)
            
            newPopulation.append(child)
        ENDWHILE
        
        currentGeneration = newPopulation
        genCount = genCount + 1
    ENDWHILE
    
    RETURN currentGeneration[0]
ENDFUNCTION

CLASS PhysicsFormulas
    private GRAVITY = 9.81
    private pressureSeaLevel = 101325
    private molarMass = 0.02897
    private R = 8.314

    PUBLIC FUNCTION downforceEquation(v)
        // Downforce is proportional to velocity squared
        RETURN v^2
    ENDFUNCTION

    // Calculates centripetal force (lateral load)
    PUBLIC FUNCTION lateral_force(m, v, r)
        // Check if the car is on a straight (radius is infinite or 0)
        IF r == 0 OR r == -1 THEN
            RETURN 0
        ELSE
            RETURN (m * (v**2)) / r
        ENDIF
    ENDFUNCTION

    // Calculates the friction limit
    PUBLIC FUNCTION max_grip_force(mu, m, downforce)
        weight = m * GRAVITY
        totalVerticalLoad = weight + downforce
        RETURN totalVerticalLoad * mu
    ENDFUNCTION

    // Linear tyre wear
    PUBLIC FUNCTION tyre_wear(lap, muStart, life)
        wearFactor = (lap * (0.2 * muStart) / life)
        currentGrip = muStart - wearFactor
         
        RETURN currentGrip
    ENDFUNCTION

    // Calculates air density using the Barometric formula
    PUBLIC FUNCTION air_density(height, temp)
        RETURN pressureSeaLevel* e ** ((-molarMass * GRAVITY * height)/ (R * temp))
    ENDFUNCTION

    PUBLIC FUNCTION drag(v, density, dragCoeff, area)
        RETURN 0.5 * dragCoeff * area * density * POW(v, 2)
    ENDFUNCTION

    PUBLIC FUNCTION max_thrust(v, power)
        IF v <= 0 THEN
            RETURN 99999 // prevent division by 0 error
        ENDIF        
        RETURN power / v
    ENDFUNCTION
ENDCLASS

FUNCTION Velocity(maxVelArr, mass, density)
    velocities = []
    
    // Forward Pass acceleration and top speed limits
    FOR i = 0 TO length(maxVelArr) - 1
        u = velocities[i - 1]
        // Physics: Resultant Force = Thrust - Drag
        force = CalculateThrust(u) - CalculateDrag(u, density)
        acceleration = force / mass
        
        // v = u + at
        newVel = u + (acceleration * 0.01)
        velocities[i] = MIN(maxVelArr[i], newVel)
    NEXT i

    // Corrects velocity if the car cannot slow down in time for a corner
    FOR i = length(velocities) - 2 TO 0 STEP -1
        maxDecel = 7 * 9.61 //7g's
        IF velocities[i] > (velocities[i + 1] + maxDecel) THEN
            velocities[i] = velocities[i + 1] + maxDecel
        ENDIF
    NEXT i

    RETURN velocities
ENDFUNCTION
